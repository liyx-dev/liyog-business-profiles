/* =====================================================================
   LIYOG WORLD — auth-ui.js
   Handles Google Sign-In, the signup form (name/category/slug/consent),
   and session state. Loaded only on states that need auth — the claim
   screen and, later, the dashboard — not on every profile view.
   ===================================================================== */

(function () {
  "use strict";

  const WORKER_BASE = "https://www.liyogworld.com.ng";
  let GOOGLE_CLIENT_ID = null;

  window.LiyogAuth = {
    init,
    getCurrentUser,
    renderSignInButton,
    renderSignupForm,
    logout
  };

  let cachedUser = null;

  async function init() {
    const cfgRes = await fetch(`${WORKER_BASE}/api/config`);
    const cfg = await cfgRes.json();
    GOOGLE_CLIENT_ID = cfg.google_client_id;
    await loadGoogleScript();
  }

  function loadGoogleScript() {
    return new Promise((resolve) => {
      if (window.google && window.google.accounts) { resolve(); return; }
      const script = document.createElement("script");
      script.src = "https://accounts.google.com/gsi/client";
      script.onload = resolve;
      document.head.appendChild(script);
    });
  }

  async function getCurrentUser() {
    if (cachedUser) return cachedUser;
    try {
      const res = await fetch(`${WORKER_BASE}/api/auth/me`, { credentials: "include" });
      const data = await res.json();
      cachedUser = data.loggedIn ? data : { loggedIn: false };
      return cachedUser;
    } catch (e) {
      return { loggedIn: false };
    }
  }

  async function logout() {
    await fetch(`${WORKER_BASE}/api/auth/logout`, { method: "POST", credentials: "include" });
    cachedUser = null;
    window.location.reload();
  }

  /**
   * Renders the Google Sign-In button into the given container element.
   * onSuccess receives { user, profiles } once sign-in completes.
   */
  function renderSignInButton(containerEl, onSuccess) {
    if (!window.google || !GOOGLE_CLIENT_ID) {
      containerEl.innerHTML = `<p class="lp-auth-error">Sign-in is temporarily unavailable. Please try again shortly.</p>`;
      return;
    }

    window.google.accounts.id.initialize({
      client_id: GOOGLE_CLIENT_ID,
      callback: async (response) => {
        containerEl.innerHTML = renderAuthLoadingState();
        try {
          const marketingOptIn = document.getElementById("lp-consent-checkbox")?.checked || false;
          const res = await fetch(`${WORKER_BASE}/api/auth/google`, {
            method: "POST",
            credentials: "include",
            headers: { "content-type": "application/json" },
            body: JSON.stringify({ credential: response.credential, marketingOptIn })
          });
          if (!res.ok) throw new Error("Sign-in failed");
          const data = await res.json();
          cachedUser = { loggedIn: true, user: data.user, profiles: data.profiles };
          onSuccess(cachedUser);
        } catch (err) {
          console.error("Sign-in error:", err);
          containerEl.innerHTML = `<p class="lp-auth-error">Something went wrong signing you in. Please try again.</p>`;
        }
      }
    });

    window.google.accounts.id.renderButton(containerEl, {
      theme: "outline",
      size: "large",
      width: 280,
      text: "continue_with",
      shape: "pill"
    });
  }

  function renderAuthLoadingState() {
    return `<div class="lp-auth-loading"><div class="lp-auth-spinner"></div><span>Signing you in…</span></div>`;
  }

  /**
   * Renders the full signup form: consent checkbox + Google button first
   * (consent must exist before sign-in fires, since the callback reads it),
   * then after sign-in succeeds, the brand-details step (name, category,
   * slug with live availability).
   */
  function renderSignupForm(containerEl) {
    containerEl.innerHTML = `
      <div class="lp-auth-card">
        <h2>Create Your Brand Profile</h2>
        <p class="lp-auth-sub">Sign in with Google to get started — takes less than a minute.</p>
        <label class="lp-consent-row">
          <input type="checkbox" id="lp-consent-checkbox">
          <span>I'd like to receive relevant updates and offers from Liyog World.</span>
        </label>
        <div id="lp-google-btn-slot"></div>
        <p class="lp-auth-fineprint">By continuing, you agree to our use of your name, email, and photo to set up your account.</p>
      </div>
    `;

    const btnSlot = document.getElementById("lp-google-btn-slot");
    renderSignInButton(btnSlot, (authState) => {
      if (authState.profiles && authState.profiles.length > 0) {
        renderDashboardRedirect(containerEl, authState);
      } else {
        renderBrandDetailsStep(containerEl, authState);
      }
    });
  }

  function renderDashboardRedirect(containerEl, authState) {
    containerEl.innerHTML = `
      <div class="lp-auth-card">
        <h2>Welcome back, ${escHtml(authState.user.name)}</h2>
        <p class="lp-auth-sub">You already have ${authState.profiles.length} brand profile${authState.profiles.length > 1 ? "s" : ""}.</p>
        <div class="lp-existing-profiles">
          ${authState.profiles.map((p) => `
            <a class="lp-existing-profile-link" href="${WORKER_BASE}/b/${escAttr(p.slug)}">
              ${escHtml(p.business_name || p.slug)}
              <span class="lp-status-badge lp-status-${p.moderation_status}">${p.moderation_status}</span>
            </a>
          `).join("")}
        </div>
        <button class="lp-btn lp-btn-secondary" id="lp-add-another-btn" style="margin-top:16px;width:100%;">
          <span>Add another brand profile</span>
        </button>
      </div>
    `;
    document.getElementById("lp-add-another-btn").addEventListener("click", () => {
      renderBrandDetailsStep(containerEl, authState);
    });
  }

  async function renderBrandDetailsStep(containerEl, authState) {
    let categories = [];
    try {
      const res = await fetch(`${WORKER_BASE}/api/categories`);
      const data = await res.json();
      categories = data.categories || [];
    } catch (e) { /* form still works with empty dropdown, shows a message */ }

    containerEl.innerHTML = `
      <div class="lp-auth-card">
        <div class="lp-welcome-row">
          <img src="${escAttr(authState.user.avatar)}" alt="" class="lp-welcome-avatar">
          <span>Signed in as <b>${escHtml(authState.user.name)}</b></span>
        </div>
        <h2>Tell Us About Your Brand</h2>
        <form id="lp-brand-form">
          <label class="lp-field-label">Brand name</label>
          <input type="text" id="lp-input-name" class="lp-field-input" placeholder="e.g. Zion Store" required maxlength="60">

          <label class="lp-field-label">Category</label>
          <select id="lp-input-category" class="lp-field-input" required>
            <option value="">Choose a category</option>
            ${categories.map((c) => `<option value="${escAttr(c.slug)}">${escHtml(c.label)}</option>`).join("")}
          </select>

          <label class="lp-field-label">Your unique link</label>
          <div class="lp-slug-input-wrap">
            <span class="lp-slug-prefix">liyogworld.com.ng/b/</span>
            <input type="text" id="lp-input-slug" class="lp-field-input lp-slug-input" placeholder="yourbrand" required maxlength="20">
          </div>
          <p class="lp-slug-status" id="lp-slug-status"></p>

          <label class="lp-field-label">Short tagline (optional)</label>
          <input type="text" id="lp-input-tagline" class="lp-field-input" placeholder="e.g. Quality goods, honest prices" maxlength="100">

          <button type="submit" class="lp-btn lp-btn-primary lp-submit-btn" id="lp-submit-btn" disabled>
            <span>Create My Brand Profile</span>
          </button>
        </form>
      </div>
    `;

    const nameInput = document.getElementById("lp-input-name");
    const slugInput = document.getElementById("lp-input-slug");
    const statusEl = document.getElementById("lp-slug-status");
    const submitBtn = document.getElementById("lp-submit-btn");
    let slugAvailable = false;
    let slugCheckTimer = null;
    let userEditedSlug = false;

    // Auto-suggest a slug from the brand name, but stop once the user
    // starts typing their own — never fight the person's own input.
    nameInput.addEventListener("input", () => {
      if (!userEditedSlug) {
        slugInput.value = slugify(nameInput.value);
        checkSlugAvailability(slugInput.value);
      }
    });

    slugInput.addEventListener("input", () => {
      userEditedSlug = true;
      slugInput.value = slugify(slugInput.value);
      checkSlugAvailability(slugInput.value);
    });

    function checkSlugAvailability(slug) {
      clearTimeout(slugCheckTimer);
      if (!slug) { statusEl.textContent = ""; slugAvailable = false; updateSubmitState(); return; }
      statusEl.textContent = "Checking availability…";
      statusEl.className = "lp-slug-status lp-slug-checking";
      slugCheckTimer = setTimeout(async () => {
        try {
          const res = await fetch(`${WORKER_BASE}/api/check-slug?slug=${encodeURIComponent(slug)}`);
          const data = await res.json();
          slugAvailable = data.available;
          if (data.available) {
            statusEl.textContent = "This link is available";
            statusEl.className = "lp-slug-status lp-slug-ok";
          } else {
            statusEl.textContent = data.reason === "taken" ? "This link is already taken" : "Please choose a different link";
            statusEl.className = "lp-slug-status lp-slug-taken";
          }
        } catch (e) {
          statusEl.textContent = "Couldn't check availability, try again";
          statusEl.className = "lp-slug-status lp-slug-taken";
        }
        updateSubmitState();
      }, 450);
    }

    function updateSubmitState() {
      submitBtn.disabled = !(nameInput.value.trim() && document.getElementById("lp-input-category").value && slugAvailable);
    }

    [nameInput, document.getElementById("lp-input-category")].forEach((el) => {
      el.addEventListener("input", updateSubmitState);
      el.addEventListener("change", updateSubmitState);
    });

    document.getElementById("lp-brand-form").addEventListener("submit", async (e) => {
      e.preventDefault();
      submitBtn.disabled = true;
      submitBtn.innerHTML = "<span>Creating your profile…</span>";

      try {
        const res = await fetch(`${WORKER_BASE}/api/profiles`, {
          method: "POST",
          credentials: "include",
          headers: { "content-type": "application/json" },
          body: JSON.stringify({
            business_name: nameInput.value.trim(),
            business_category: document.getElementById("lp-input-category").value,
            slug: slugInput.value,
            tagline: document.getElementById("lp-input-tagline").value.trim() || null
          })
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error || "Something went wrong");

        renderSuccessState(containerEl, data.slug, data.moderationStatus);
      } catch (err) {
        submitBtn.disabled = false;
        submitBtn.innerHTML = "<span>Create My Brand Profile</span>";
        statusEl.textContent = err.message;
        statusEl.className = "lp-slug-status lp-slug-taken";
      }
    });
  }

  function renderSuccessState(containerEl, slug, moderationStatus) {
    const isApproved = moderationStatus === "approved";
    containerEl.innerHTML = `
      <div class="lp-auth-card lp-auth-success">
        <div class="lp-success-icon">
          <svg width="40" height="40" viewBox="0 0 24 24" fill="none" stroke="#25D366" stroke-width="2.5"><polyline points="20 6 9 17 4 12"/></svg>
        </div>
        <h2>${isApproved ? "You're Live!" : "Almost There"}</h2>
        <p class="lp-auth-sub">
          ${isApproved
            ? "Your brand profile has been created and is now live."
            : "Your profile has been created and is being reviewed. This usually takes a few hours."}
        </p>
        <a class="lp-state-cta" href="${WORKER_BASE}/b/${escAttr(slug)}" style="margin-top:8px;">
          View Your Profile
        </a>
      </div>
    `;
  }

  function slugify(str) {
    return str.toLowerCase().trim()
      .replace(/[^a-z0-9\s-]/g, "")
      .replace(/\s+/g, "-")
      .replace(/-+/g, "-")
      .slice(0, 20)
      .replace(/^-+|-+$/g, "");
  }

  function escHtml(str) {
    if (!str) return "";
    const div = document.createElement("div");
    div.textContent = str;
    return div.innerHTML;
  }
  function escAttr(str) {
    if (!str) return "";
    return String(str).replace(/"/g, "&quot;");
  }
})();
