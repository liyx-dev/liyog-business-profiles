(function () {
  "use strict";

  const WORKER_BASE = "https://www.liyogworld.com.ng";
  const TEMPLATE_URL = `${WORKER_BASE}/brands-template.html`;
  const BIO_PREVIEW_CHAR_LIMIT = 160;

  const mount = document.getElementById("liyog-profile-root");
  if (!mount) { console.error("liyog-profile-root not found."); return; }

  function getOverlayHost() {
    return document.querySelector(".lp-root") || mount || document.body;
  }

  const params = new URLSearchParams(window.location.search);
  const slug = params.get("biz");
  const claimSlug = params.get("claim");
  let pagePath = window.location.pathname;

  init();

  async function init() {
    try {
      const cfgRes = await fetch(`${WORKER_BASE}/api/config`);
      const cfg = await cfgRes.json();
      if (cfg.blogger_profile_page) pagePath = cfg.blogger_profile_page;
    } catch (e) { /* fallback already set */ }

    if (claimSlug) { showClaimScreen(claimSlug); return; }
    if (!slug) { showClaimScreen(null); return; }

    showSkeleton();

    let templateHtml;
    try {
      const tplRes = await fetch(TEMPLATE_URL);
      templateHtml = await tplRes.text();
    } catch (e) {
      console.error("Template fetch failed:", e);
      showErrorScreen();
      return;
    }

    try {
      const currentUserId = await getCurrentUserId();
      const dataRes = await fetch(
        `${WORKER_BASE}/b/${encodeURIComponent(slug)}?format=json${currentUserId ? `&viewer_id=${encodeURIComponent(currentUserId)}` : ""}`
      );
      const data = await dataRes.json();

      if (!data.found) { showNotFoundScreen(slug); return; }
      if (data.status === "pending_review") { showPendingScreen(); return; }

      mount.innerHTML = templateHtml;
      const isOwner = currentUserId && currentUserId === data.profile.owner_id;
      bindProfile(data.profile, isOwner);
      injectSeoTags(data.profile);
      cleanAddressBar(slug);
    } catch (e) {
      console.error("Profile fetch/render failed:", e);
      showErrorScreen();
    }
  }

  function bindProfile(p, isOwner) {
    const DEFAULT_COVER = "https://placehold.co/1200x560/3F3D9E/FFFFFF?text=Add+Your+Cover+Photo";
    const DEFAULT_LOGO = "https://placehold.co/300x300/1877F2/FFFFFF?text=Logo";

    setImg("[data-bind='cover_url']", p.cover_url || DEFAULT_COVER, p.cover_url ? `${p.business_name} cover` : "Cover photo not yet added");
    setImg(".lp-logo", p.logo_url || DEFAULT_LOGO, p.logo_url ? `${p.business_name} logo` : "Logo not yet added");
    setText("[data-bind='business_name']", p.business_name);
    setText("[data-bind='tagline']", p.tagline || "");
    if (!p.tagline) hide(".lp-tagline");

    const verified = document.getElementById("lp-verified");
    if (verified && p.moderation_status === "approved" && p.is_verified) verified.hidden = false;

    const editBtn = document.getElementById("lp-edit-toggle");
    if (editBtn && isOwner) {
      editBtn.hidden = false;
      editBtn.addEventListener("click", () => openEditPanel(p));
    }

    if (isOwner && (!p.logo_url || !p.cover_url || !p.bio_html)) {
      showCompletionNudge(p);
    }

    bindPills(safeParseArray(p.key_points));
    bindCategoryBadge(p.business_category);
    bindContactBanner(p);
    bindActions(p);
    bindInquiry(p);
    bindBio(p);
    bindGallery(safeParseArray(p.store_photos), p.business_name);
    bindProducts(p, isOwner); // <--- ADDED PRODUCT BINDING HERE
    bindReviews(p, isOwner);
    bindVideo(p.youtube_url, p.business_name);
    bindMap(p);
    bindSocial(p);
    bindMeta(p);
    wireShare(p);
    wireLightbox();
    wireSheets();
  }

    retail: "Retail & Shops", fashion: "Fashion & Apparel", food: "Food & Restaurants",
    beauty: "Beauty & Personal Care", electronics: "Electronics & Gadgets",
    services: "Professional Services", health: "Health & Wellness",
    education: "Education & Training", automotive: "Automotive",
    real_estate: "Real Estate", agriculture: "Agriculture",
    events: "Events & Entertainment", logistics: "Logistics & Delivery", other: "Other"

  function bindCategoryBadge(categorySlug) {
    const badge = document.getElementById("lp-category-badge");
    if (!badge || !categorySlug) return;
    badge.textContent = CATEGORY_LABELS[categorySlug] || categorySlug;

  function bindContactBanner(p) {
    const banner = document.getElementById("lp-contact-banner");
    if (!banner) return;
    let any = false;

    if (p.whatsapp_number) {
      document.getElementById("lp-contact-whatsapp-row").hidden = false;
      document.getElementById("lp-contact-whatsapp-value").textContent = formatDisplayNumber(p.whatsapp_number);
      any = true;
    }
    if (p.phone_number) {
      document.getElementById("lp-contact-phone-row").hidden = false;
      document.getElementById("lp-contact-phone-value").textContent = formatDisplayNumber(p.phone_number);
      any = true;
    }
    if (p.response_time) {
      document.getElementById("lp-contact-response-row").hidden = false;
      document.getElementById("lp-contact-response-value").textContent = p.response_time;
      any = true;
    }
    if (any) banner.hidden = false;
  }

  function formatDisplayNumber(number) {
    const cleaned = number.replace(/[^\d+]/g, "");
    return cleaned.startsWith("+") ? cleaned : `+${cleaned}`;
  }

  function bindPills(points) {
    const wrap = document.getElementById("lp-pills");
    if (!wrap || !points.length) return;
    points.slice(0, 5).forEach((text) => {
      const pill = document.createElement("span");
      pill.className = "lp-pill";
      const icon = document.createElementNS("http://www.w3.org/2000/svg", "svg");
      icon.setAttribute("width", "12"); icon.setAttribute("height", "12");
      icon.setAttribute("viewBox", "0 0 24 24"); icon.setAttribute("fill", "none");
      icon.setAttribute("stroke", "currentColor"); icon.setAttribute("stroke-width", "3");
      icon.innerHTML = '<polyline points="20 6 9 17 4 12"/>';
      pill.appendChild(icon);
      pill.appendChild(document.createTextNode(text));
      wrap.appendChild(pill);
    });
  }

    const waBtn = document.getElementById("lp-btn-whatsapp");
    if (waLink && waBtn) { waBtn.href = waLink; waBtn.hidden = false; }

    if (p.phone_number && callBtn) { callBtn.href = `tel:${sanitizeTel(p.phone_number)}`; callBtn.hidden = false; }

    if (p.phone_number && smsBtn) {

      smsBtn.href = `sms:${sanitizeTel(p.phone_number)}?body=${encodeURIComponent(smsText)}`;
      smsBtn.hidden = false;
    }
  }

  function bindBio(p) {
    if (!p.bio_html) return;
    const section = document.getElementById("lp-about-section");
    const preview = document.getElementById("lp-bio-preview");
    const fade = document.getElementById("lp-bio-fade");
    const readMoreBtn = document.getElementById("lp-read-more-btn");
    if (!section || !preview) return;

    const cleanHtml = sanitizeBioHtml(p.bio_html);
    const plainText = stripHtml(cleanHtml);
    preview.innerHTML = cleanHtml;
    section.hidden = false;

    const overflows = plainText.length > BIO_PREVIEW_CHAR_LIMIT;
    if (overflows) {
      readMoreBtn.hidden = false;
      if (fade) fade.style.display = "block";
      readMoreBtn.addEventListener("click", () => openBioSheet(p, cleanHtml));
    } else {
      if (fade) fade.style.display = "none";
      preview.style.webkitLineClamp = "unset";
      preview.style.maxHeight = "none";
    }
  }

    const overlay = document.getElementById("lp-bio-sheet-overlay");
    const logo = document.getElementById("lp-sheet-logo");
    const name = document.getElementById("lp-sheet-name");
    const body = document.getElementById("lp-sheet-bio");
    const ctaRow = document.getElementById("lp-sheet-cta-row");
    if (!overlay) return;

    logo.src = p.logo_url || "";
    logo.alt = `${p.business_name} logo`;
    name.textContent = p.business_name;
    body.innerHTML = cleanBioHtml;

    ctaRow.innerHTML = "";
    const waLink = buildWhatsAppLink(p.whatsapp_number, p.wa_message, p.business_name);
    if (waLink) {
      const a = document.createElement("a");
      a.className = "lp-btn lp-btn-primary";
      a.href = waLink; a.target = "_blank"; a.rel = "noopener";
      a.innerHTML = "<span>WhatsApp</span>";
      ctaRow.appendChild(a);

    if (p.phone_number) {
      const a = document.createElement("a");

      a.href = `tel:${sanitizeTel(p.phone_number)}`;
      a.innerHTML = "<span>Call</span>";
      ctaRow.appendChild(a);

  }

    if (!photos.length) return;
    const section = document.getElementById("lp-gallery-section");
    const wrap = document.getElementById("lp-gallery");
    if (!section || !wrap) return;
    wrap.innerHTML = ""; // Clear out previous items before binding
    photos.slice(0, 5).forEach((url, i) => {
      const tile = document.createElement("div");
      tile.className = "lp-gallery-tile";
      const img = document.createElement("img");
      img.src = url;
      img.loading = "lazy";
      img.alt = `${businessName} — brand photo ${i + 1}`;
      img.dataset.full = url;
      tile.appendChild(img);
      const index = document.createElement("span");
      index.className = "lp-gallery-index";
      index.textContent = `${i + 1} / ${photos.length}`;
      tile.appendChild(index);
      wrap.appendChild(tile);
    });
    section.hidden = false;
  }

  }

  function wireEditSlugField(currentSlug) {
    const input = document.getElementById("lp-edit-slug-input");
    const statusEl = document.getElementById("lp-edit-slug-status");
    if (!input || !statusEl) return;

    let checkTimer = null;
    input.addEventListener("input", () => {
      input.value = input.value.toLowerCase().replace(/[^a-z0-9-]/g, "");
      clearTimeout(checkTimer);
      const slug = input.value;

      if (slug === currentSlug) { statusEl.textContent = ""; return; }
      if (!slug) { statusEl.textContent = ""; return; }

      statusEl.textContent = "Checking availability…";
      statusEl.className = "lp-slug-status lp-slug-checking";
      checkTimer = setTimeout(async () => {
        try {
          const res = await fetch(`${WORKER_BASE}/api/check-slug?slug=${encodeURIComponent(slug)}`);
          const data = await res.json();
          if (data.available) {
            statusEl.textContent = "This link is available";
            statusEl.className = "lp-slug-status lp-slug-ok";
          } else {
            statusEl.textContent = data.reason === "taken" ? "This link is already taken" : "Please choose a different link";
            statusEl.className = "lp-slug-status lp-slug-taken";
          }
        } catch (e) {
          statusEl.textContent = "Couldn't check availability right now";
          statusEl.className = "lp-slug-status lp-slug-taken";
        }
      }, 450);
    });
  }

  // UPDATED WITH PRODUCTS TAB LOGIC
  function wireEditTabs(p) {
    const detailsTab = document.getElementById("lp-tab-details");
    const inquiriesTab = document.getElementById("lp-tab-inquiries");
    
    const detailsPanel = document.getElementById("lp-tab-panel-details");
    const inquiriesPanel = document.getElementById("lp-tab-panel-inquiries");
    
    if (!detailsTab || !inquiriesTab) return;

    detailsTab.onclick = () => {
      detailsTab.classList.add("lp-edit-tab-active");
      inquiriesTab.classList.remove("lp-edit-tab-active");
      detailsPanel.hidden = false;
      inquiriesPanel.hidden = true;
    };

    inquiriesTab.onclick = () => {
      inquiriesTab.classList.add("lp-edit-tab-active");
      detailsTab.classList.remove("lp-edit-tab-active");
      inquiriesPanel.hidden = false;
      detailsPanel.hidden = true;
      loadInquiriesForOwner(p.id);
    };

    detailsTab.classList.add("lp-edit-tab-active");
    inquiriesTab.classList.remove("lp-edit-tab-active");
    detailsPanel.hidden = false;
    inquiriesPanel.hidden = true;
  }

  async function loadInquiriesForOwner(profileId) {
    const listEl = document.getElementById("lp-inquiries-list");
    const badgeEl = document.getElementById("lp-inquiries-count");
    if (!listEl) return;
    listEl.innerHTML = `<p class="lp-inquiry-empty">Loading…</p>`;

    try {
      const res = await fetch(`${WORKER_BASE}/api/profiles/${profileId}/inquiries`, { credentials: "include" });
      const data = await res.json();
      if (!res.ok) throw new Error(data.error || "Couldn't load inquiries");

      const inquiries = data.inquiries || [];
      if (badgeEl) {
        if (inquiries.length > 0) { badgeEl.textContent = inquiries.length; badgeEl.hidden = false; }
        else { badgeEl.hidden = true; }
      }

      if (!inquiries.length) {
        listEl.innerHTML = `<p class="lp-inquiry-empty">No inquiries yet. When someone sends you a message, it'll show up here. Messages are kept for 7 days.</p>`;
        return;
      }

      listEl.innerHTML = inquiries.map((inq) => `
        <div class="lp-inquiry-list-item">
          <div class="lp-inquiry-list-top">
            <span class="lp-inquiry-list-name">${escHtml(inq.sender_name)}</span>
            <span class="lp-inquiry-list-date">${formatExactDate(inq.created_at)}</span>
          </div>
          <div class="lp-inquiry-list-contact">
            ${inq.action_url 
              ? `<a href="${escAttr(inq.action_url)}" target="_blank" class="lp-inquiry-action-link">${escHtml(inq.sender_contact)} ↗</a>`
              : escHtml(inq.sender_contact)
            }
          </div>
          <p class="lp-inquiry-list-message">${escHtml(inq.message)}</p>
        </div>
      `).join("");
    } catch (err) {
      listEl.innerHTML = `<p class="lp-inquiry-empty">${escHtml(err.message)}</p>`;
    }
  }

  async function wireCategoryDropdown(currentCategory) {
    const select = document.getElementById("lp-edit-category-select");
    if (!select) return;
    try {
      const res = await fetch(`${WORKER_BASE}/api/categories`);
      const data = await res.json();
      select.innerHTML = '<option value="">Choose a category</option>';
      (data.categories || []).forEach((c) => {
        const opt = document.createElement("option");
        opt.value = c.slug;
        opt.textContent = c.label;
        if (c.slug === currentCategory) opt.selected = true;
        select.appendChild(opt);
      });
    } catch (e) {
      console.error("Failed to load categories:", e);
    }
  }

  function wireTagInput(initialTags) {
    const box = document.getElementById("lp-tag-input-box");
    const chipRow = document.getElementById("lp-tag-chip-row");
    const textInput = document.getElementById("lp-tag-text-input");
    const hiddenInput = document.getElementById("lp-hidden-key-points");
    if (!box || !chipRow || !textInput) return;

    let tags = [...initialTags].slice(0, 5);

    box.addEventListener("click", () => textInput.focus());

    textInput.addEventListener("keydown", (e) => {
      if (e.key === "Enter" || e.key === ",") {
        e.preventDefault();
        addTagFromInput();
      } else if (e.key === "Backspace" && !textInput.value && tags.length) {
        tags.pop();
        syncTags();
      }
    });
    textInput.addEventListener("blur", addTagFromInput);

    function addTagFromInput() {
      const value = textInput.value.trim();
      if (!value || tags.length >= 5) { textInput.value = ""; return; }
      tags.push(value.slice(0, 40));
      textInput.value = "";
      syncTags();
    }

    function removeTag(index) {
      tags.splice(index, 1);
      syncTags();
    }

    function syncTags() {
      renderChips();
      hiddenInput.value = JSON.stringify(tags);
    }

    function renderChips() {
      chipRow.innerHTML = "";
      tags.forEach((tag, i) => {
        const chip = document.createElement("span");
        chip.className = "lp-tag-chip";
        chip.innerHTML = `<span>${escHtml(tag)}</span>`;
        const removeBtn = document.createElement("button");
        removeBtn.type = "button";
        removeBtn.className = "lp-tag-chip-remove";
        removeBtn.innerHTML = `<svg width="10" height="10" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>`;
        removeBtn.addEventListener("click", () => removeTag(i));
        chip.appendChild(removeBtn);
        chipRow.appendChild(chip);
      });
      textInput.disabled = tags.length >= 5;
      textInput.placeholder = tags.length >= 5 ? "Maximum 5 reached" : "Type a highlight and press Enter";
    }

    syncTags();
  }

  function wireGalleryUpload(form, p) {
    const grid = document.getElementById("lp-gallery-upload-grid");
    if (!grid) return;
    let photos = safeParseArray(p.store_photos).slice(0, 5);
    while (photos.length < 5) photos.push(null);

    renderGalleryGrid();

    function renderGalleryGrid() {
      grid.innerHTML = photos.map((url, i) => `
        <div class="lp-gallery-upload-slot" data-index="${i}">
          ${url
            ? `<img src="${escAttr(url)}" alt="">
               <button type="button" class="lp-gallery-remove-btn" data-index="${i}" aria-label="Remove photo">

          openFilePickerForSlot(idx);
        });

        btn.addEventListener("click", async (e) => {
          e.stopPropagation();

          if (!oldUrl) return;

          const slot = btn.closest(".lp-gallery-upload-slot");

          statusEl.textContent = "Removing…";
          statusEl.className = "lp-gallery-slot-status lp-upload-active";

            const res = await fetch(`${WORKER_BASE}/api/remove-image?field=store_photos&profile_id=${p.id}&url=${encodeURIComponent(oldUrl)}`, {
              method: "POST",
              credentials: "include"
            });
            const data = await res.json();
            if (!res.ok) throw new Error(data.error || "Removal failed");

            photos[idx] = null;
            syncHiddenInput();
            renderGalleryGrid();
            bindGallery(photos.filter(Boolean), p.business_name);

    function openFilePickerForSlot(idx) {
      const fileInput = document.createElement("input");
      fileInput.type = "file";
      fileInput.accept = "image/*";
      fileInput.addEventListener("change", async () => {
        const file = fileInput.files[0];
        if (!file) return;
        const statusEl = grid.querySelector(`.lp-gallery-slot-status[data-index="${idx}"]`);
        statusEl.textContent = "Processing…";
        statusEl.className = "lp-gallery-slot-status lp-upload-active";
        try {
          const webpBlob = await compressImageToWebp(file, 900);
          statusEl.textContent = "Uploading…";
          const filename = buildSeoFilename(p.business_name, `gallery-${idx + 1}`);

          const existingUrl = photos[idx];
          const replaceParam = existingUrl ? `&replace_url=${encodeURIComponent(existingUrl)}` : "";

          const res = await fetch(`${WORKER_BASE}/api/upload-image?name=${encodeURIComponent(filename)}&field=store_photos&profile_id=${p.id}${replaceParam}`, {
            method: "POST",
            credentials: "include",
            headers: { "content-type": "image/webp" },
            body: webpBlob
          });
          const data = await res.json();
          if (!res.ok) throw new Error(data.error || "Upload failed");
          
          photos[idx] = data.url;
          syncHiddenInput();
          renderGalleryGrid();
          bindGallery(photos.filter(Boolean), p.business_name);
        } catch (err) {
          statusEl.textContent = err.message;
          statusEl.className = "lp-gallery-slot-status lp-upload-error";
          showToast(err.message, "error");
        }
      });
      fileInput.click();
    }

    function syncHiddenInput() {
      let hidden = form.querySelector("input[name='store_photos']");
      if (!hidden) {
        hidden = document.createElement("input");
        hidden.type = "hidden";
        hidden.name = "store_photos";

    syncHiddenInput();
  }

  function buildSeoFilename(businessName, fieldType) {
    const slugified = (businessName || "brand")
      .toLowerCase()
      .replace(/[^a-z0-9\s-]/g, "")
      .replace(/\s+/g, "-")
      .slice(0, 40);
    return `${slugified}-${fieldType}`;
  }

  const HEX_TO_COLOR_NAME = {
    "#ff3b30": "red",
    "#28a428": "green",
    "#1877f2": "blue",
    "#ffd700": "gold",
    "#111111": "black"
  };

  function htmlToRichTextSyntax(html) {
    if (!html) return "";
    let syntax = html;
    syntax = syntax.replace(/<a href="([^"]+)"[^>]*>([^<]*)<\/a>/g, "[$2]($1)");
    syntax = syntax.replace(/<span style="color:(#[0-9A-Fa-f]{6})">([^<]*)<\/span>/g, (match, hex, text) => {
      const colorName = HEX_TO_COLOR_NAME[hex.toLowerCase()] || "gold";
      return `{{${colorName}:${text}}}`;
    });
    syntax = syntax.replace(/<strong>([^<]*)<\/strong>/g, "*$1*");
    syntax = syntax.replace(/<em>([^<]*)<\/em>/g, "_$1_");
    syntax = syntax.replace(/<u>([^<]*)<\/u>/g, "~$1~");
    syntax = syntax.replace(/<br\s*\/?>/g, "\n");
    const tmp = document.createElement("div");
    tmp.innerHTML = syntax;
    return tmp.textContent || "";
  }

  function wireRichTextToolbar() {

    const textarea = document.getElementById("lp-bio-textarea");
    const charCount = document.getElementById("lp-bio-char-count");
    if (!toolbar || !textarea) return;

    const helpBtn = document.getElementById("lp-richtext-help-btn");
    if (helpBtn) helpBtn.addEventListener("click", openFormattingHelpSheet);

    function wrapSelection(marker) {
      const start = textarea.selectionStart;
      const end = textarea.selectionEnd;
      const selected = textarea.value.slice(start, end) || "text";
      const before = textarea.value.slice(0, start);
      const after = textarea.value.slice(end);
      textarea.value = `${before}${marker}${selected}${marker}${after}`;
      textarea.focus();
      textarea.setSelectionRange(start + marker.length, start + marker.length + selected.length);
      updateCharCount();
    }

    function wrapColor(colorName) {
      const start = textarea.selectionStart;
      const end = textarea.selectionEnd;
      const selected = textarea.value.slice(start, end) || "text";
      const before = textarea.value.slice(0, start);
      const after = textarea.value.slice(end);
      const wrapped = `{{${colorName}:${selected}}}`;
      textarea.value = `${before}${wrapped}${after}`;
      textarea.focus();
      updateCharCount();
    }

    function updateCharCount() {
      if (charCount) charCount.textContent = `${textarea.value.length} / 3000 characters`;
    }

    toolbar.querySelectorAll(".lp-rt-btn[data-wrap]").forEach((btn) => {
      btn.addEventListener("click", () => wrapSelection(btn.dataset.wrap));
    });
    toolbar.querySelectorAll(".lp-rt-color-btn[data-color]").forEach((btn) => {
      btn.addEventListener("click", () => wrapColor(btn.dataset.color));
    });
    document.getElementById("lp-rt-link-btn")?.addEventListener("click", () => {
      const url = prompt("Paste the link URL (must start with https://)");
      if (!url || !/^https?:\/\//i.test(url)) {
        if (url) alert("Links must start with http:// or https://");
        return;
      }
      const start = textarea.selectionStart;
      const end = textarea.selectionEnd;
      const selected = textarea.value.slice(start, end) || "link text";
      const before = textarea.value.slice(0, start);
      const after = textarea.value.slice(end);
      textarea.value = `${before}[${selected}](${url})${after}`;
      textarea.focus();
      updateCharCount();
    });

    textarea.addEventListener("input", updateCharCount);
    updateCharCount();
  }

  function slugFieldHtml(p) {
    let daysLeft = 0;
    if (p.slug_updated_at) {
      const daysSinceChange = (Date.now() - new Date(p.slug_updated_at.replace(" ", "T") + "Z").getTime()) / (1000 * 60 * 60 * 24);
      daysLeft = Math.max(0, Math.ceil(7 - daysSinceChange));
    }
    const isLocked = daysLeft > 0;

    if (isLocked) {
      return `
        <div class="lp-edit-field">
          <label class="lp-field-label">Your link</label>
          <div class="lp-slug-locked-box">
            <span>liyogworld.com.ng/b/${escHtml(p.slug)}</span>
            <span class="lp-slug-locked-badge">🔒 ${daysLeft} day${daysLeft === 1 ? "" : "s"} left</span>
          </div>
          <p class="lp-field-hint">Your link can be changed once every 7 days. You changed it recently, so it's locked for now.</p>
        </div>
      `;
    }

    return `
      <div class="lp-edit-field">
        <label class="lp-field-label">Your link</label>
        <div class="lp-slug-input-wrap">
          <span class="lp-slug-prefix">liyogworld.com.ng/b/</span>
          <input type="text" name="slug" id="lp-edit-slug-input" class="lp-field-input lp-slug-input" value="${escAttr(p.slug)}" maxlength="20">
        </div>
        <p class="lp-slug-status" id="lp-edit-slug-status"></p>
        <p class="lp-field-hint">You can change this once every 7 days.</p>
      </div>
    `;
  }

  function wireImageUpload(form, fieldName, p) {
    const box = document.getElementById(`lp-upload-box-${fieldName}`);
    const fileInput = document.getElementById(`lp-file-${fieldName}`);
    const preview = document.getElementById(`lp-preview-${fieldName}`);
    const placeholder = document.getElementById(`lp-placeholder-${fieldName}`);
    const statusEl = document.getElementById(`lp-upload-status-${fieldName}`);
    const hiddenInput = document.getElementById(`lp-hidden-${fieldName}`);
    const removeBtn = document.getElementById(`lp-remove-${fieldName}`);

    box.addEventListener("click", (e) => {
      if (e.target.closest(".lp-image-remove-trigger")) return;
      fileInput.click();
    });

    if (removeBtn) {
      removeBtn.addEventListener("click", async (e) => {
        e.stopPropagation();
        const oldUrl = hiddenInput.value;
        if (!oldUrl) return;

        statusEl.textContent = "Removing…";
        statusEl.className = "lp-image-upload-status lp-upload-active";

        try {
          const res = await fetch(`${WORKER_BASE}/api/remove-image?field=${fieldName}&profile_id=${p.id}&url=${encodeURIComponent(oldUrl)}`, {
            method: "POST",
            credentials: "include"
          });
          const data = await res.json();
          if (!res.ok) throw new Error(data.error || "Removal failed");

          hiddenInput.value = "";
          preview.src = "";
          preview.style.display = "none";
          placeholder.style.display = "flex";
          statusEl.textContent = "";
          removeBtn.remove();

          if (fieldName === "logo_url") setImg(".lp-logo", "https://placehold.co/300x300/1877F2/FFFFFF?text=Logo", "Logo removed");
          if (fieldName === "cover_url") setImg("[data-bind='cover_url']", "https://placehold.co/1200x560/3F3D9E/FFFFFF?text=Add+Your+Cover+Photo", "Cover photo removed");
        } catch (err) {
          statusEl.textContent = "Error";
          statusEl.className = "lp-image-upload-status lp-upload-error";
          showToast(err.message, "error");
        }
      });
    }

    fileInput.addEventListener("change", async () => {
      const file = fileInput.files[0];
      if (!file) return;

      statusEl.textContent = "Processing…";
      statusEl.className = "lp-image-upload-status lp-upload-active";

      try {
        const webpBlob = await compressImageToWebp(file, fieldName === "cover_url" ? 1200 : 500);

        statusEl.textContent = "Uploading…";
        const fieldLabel = fieldName === "cover_url" ? "cover" : "logo";
        const businessNameInput = document.getElementById("lp-input-name");
        const nameForFilename = businessNameInput ? businessNameInput.value : (form.dataset.businessName || "brand");
        const filename = buildSeoFilename(nameForFilename, fieldLabel);

        const res = await fetch(`${WORKER_BASE}/api/upload-image?name=${encodeURIComponent(filename)}&field=${fieldName}&profile_id=${p.id}`, {
          method: "POST",
          credentials: "include",
          headers: { "content-type": "image/webp" },
          body: webpBlob
        });
        const data = await res.json();
        if (!res.ok) throw new Error(data.error || "Upload failed");

        hiddenInput.value = data.url;
        preview.src = data.url;
        preview.style.display = "block";
        placeholder.style.display = "none";
        statusEl.textContent = "";

        if (!document.getElementById(`lp-remove-${fieldName}`)) {
          const remTrigger = document.createElement("button");
          remTrigger.type = "button";
          remTrigger.className = "lp-image-remove-trigger";
          remTrigger.id = `lp-remove-${fieldName}`;
          remTrigger.dataset.field = fieldName;
          remTrigger.textContent = "Remove Current";
          box.appendChild(remTrigger);
          wireImageUpload(form, fieldName, p);
        }

        if (fieldName === "logo_url") setImg(".lp-logo", data.url, `${p.business_name} logo`);
        if (fieldName === "cover_url") setImg("[data-bind='cover_url']", data.url, `${p.business_name} cover`);
      } catch (err) {
        statusEl.textContent = err.message;
        statusEl.className = "lp-image-upload-status lp-upload-error";
        showToast(err.message, "error");
      }
    });
  }

  function compressImageToWebp(file, maxDimension) {
    return new Promise((resolve, reject) => {
      const img = new Image();
      const reader = new FileReader();
      reader.onload = (e) => { img.src = e.target.result; };
      reader.onerror = () => reject(new Error("Couldn't read the selected file"));
      img.onload = () => {
        let { width, height } = img;
        if (width > maxDimension || height > maxDimension) {
          const ratio = Math.min(maxDimension / width, maxDimension / height);
          width = Math.round(width * ratio);
          height = Math.round(height * ratio);
        }
        const canvas = document.createElement("canvas");
        canvas.width = width;
        canvas.height = height;
        canvas.getContext("2d").drawImage(img, 0, 0, width, height);
        canvas.toBlob(
          (blob) => (blob ? resolve(blob) : reject(new Error("Compression failed"))),
          "image/webp",
          0.82
        );
      };
      img.onerror = () => reject(new Error("Couldn't load the selected image"));
      reader.readAsDataURL(file);
    });
  }

  function validateBeforeSubmit(updates) {
    for (const [field, limit] of Object.entries(FIELD_LIMITS)) {
      const value = updates[field];
      if (value && value.length > limit) {
        const label = FIELD_LABELS[field] || field;
        return `${label} is too long — please keep it under ${limit} characters.`;
      }
    }
    return null;
  }

  async function handleEditSubmit(e, p) {
    e.preventDefault();
    const form = e.target;
    const submitBtn = document.getElementById("lp-edit-submit-btn");
    const statusEl = document.getElementById("lp-edit-form-status");

    const formData = new FormData(form);
    const updates = {};
    for (const [key, value] of formData.entries()) {
      updates[key] = value.trim() || null;
    }

    const validationError = validateBeforeSubmit(updates);
    if (validationError) {
      statusEl.textContent = validationError;
      statusEl.className = "lp-edit-form-status lp-edit-error";
      return;
    }

    submitBtn.disabled = true;
    submitBtn.innerHTML = "<span>Saving…</span>";

    try {
      const res = await fetch(`${WORKER_BASE}/api/profiles/${p.id}`, {
        method: "PATCH",
        credentials: "include",
        headers: { "content-type": "application/json" },
        body: JSON.stringify(updates)
      });

      let data;
      try {
        data = await res.json();
      } catch (parseErr) {
        throw new Error("Something went wrong on our end. Please try again in a moment.");
      }

      if (!res.ok) {
        if (res.status === 422 && data.fields) {
          let flaggedMsg = data.error || "Inappropriate text detected.";
          for (const [field, info] of Object.entries(data.fields)) {
            flaggedMsg += ` (${FIELD_LABELS[field] || field}: "${info.term}")`;
            const targetInput = form.querySelector(`[name="${field}"]`);
            if (targetInput) {
              targetInput.style.borderColor = "#FF3B30";
              targetInput.style.boxShadow = "0 0 0 3px rgba(255, 59, 48, 0.2)";
              targetInput.addEventListener("input", () => {
                targetInput.style.borderColor = "";
                targetInput.style.boxShadow = "";
              }, { once: true });
            }
          }
          if (data.strikesRemaining !== undefined) {
            flaggedMsg += ` — ${data.strikesRemaining} strikes left before locking.`;
          }
          throw new Error(flaggedMsg);
        }
        throw new Error(data.error || "Couldn't save your changes.");
      }

      statusEl.textContent = data.moderationStatus === "approved"
        ? "Saved! Your changes are now live."
        : "Saved! Your changes are being reviewed and will appear shortly.";
      statusEl.className = "lp-edit-form-status lp-edit-success";
      submitBtn.innerHTML = "<span>Saved</span>";

      if (data.newSlug) {
        setTimeout(() => { window.location.href = `${WORKER_BASE}/b/${data.newSlug}`; }, 1500);
      } else {
        setTimeout(() => window.location.reload(), 1500);
      }
    } catch (err) {
      statusEl.textContent = err.message;
      statusEl.className = "lp-edit-form-status lp-edit-error";
      submitBtn.disabled = false;
      submitBtn.innerHTML = "<span>Save Changes</span>";
      form.addEventListener("submit", (e2) => handleEditSubmit(e2, p), { once: true });
    }
  }

  function wireShare(p) {
    const btn = document.getElementById("lp-btn-share");
    if (!btn) return;
    btn.addEventListener("click", async () => {
      const shareUrl = `${WORKER_BASE}/b/${p.slug}`;
      const shareText = p.tagline || stripHtml(p.bio_html).slice(0, 100);
      trackShare(p.id, "copy_link");
      if (navigator.share) {
        try {
          await navigator.share({ title: `${p.business_name} | Liyog World`, text: shareText, url: shareUrl });
          return;
        } catch (e) { /* cancelled, fall through */ }
      }
      navigator.clipboard.writeText(shareUrl).then(() => {
        const original = btn.innerHTML;
        btn.innerHTML = `<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><polyline points="20 6 9 17 4 12"/></svg>`;
        setTimeout(() => (btn.innerHTML = original), 1600);
      });
    });
  }

  function wireLightbox() {
    const lightbox = document.getElementById("lp-lightbox");
    const img = document.getElementById("lp-lightbox-img");
    const closeBtn = document.getElementById("lp-lightbox-close");
    if (!lightbox) return;

    const navigableImages = [];
    const logoEl = document.querySelector(".lp-logo");
    const coverEl = document.querySelector(".lp-cover");
    if (logoEl && logoEl.src) navigableImages.push({ src: logoEl.src, alt: logoEl.alt });
    if (coverEl && coverEl.src) navigableImages.push({ src: coverEl.src, alt: coverEl.alt });
    document.querySelectorAll(".lp-gallery-tile img[data-full]").forEach((el) => {
      navigableImages.push({ src: el.dataset.full, alt: el.alt });
    });

    if (!navigableImages.length) return;

    let currentIndex = 0;

    function updateNavButtons() {
      const prevBtn = document.getElementById("lp-lightbox-prev");
      const nextBtn = document.getElementById("lp-lightbox-next");
      if (!prevBtn || !nextBtn) return;
      prevBtn.style.visibility = currentIndex > 0 ? "visible" : "hidden";
      nextBtn.style.visibility = currentIndex < navigableImages.length - 1 ? "visible" : "hidden";
    }

    function showImageAt(index) {
      if (index < 0 || index >= navigableImages.length) return;
      currentIndex = index;
      img.src = navigableImages[index].src;
      img.alt = navigableImages[index].alt;
      updateNavButtons();
    }

    function openAt(index) {
      showImageAt(index);
      openSheet(lightbox);
    }

    if (logoEl) logoEl.addEventListener("click", () => openAt(0));
    if (coverEl) coverEl.addEventListener("click", () => openAt(navigableImages.length > 1 && coverEl.src !== logoEl?.src ? 1 : 0));

    const gallery = document.getElementById("lp-gallery");
    if (gallery) {
      gallery.addEventListener("click", (e) => {
        const target = e.target.closest(".lp-gallery-tile img[data-full]");
        if (!target) return;
        const galleryIndex = navigableImages.findIndex((item) => item.src === target.dataset.full);
        openAt(galleryIndex >= 0 ? galleryIndex : 0);
      });
    }

    closeBtn.addEventListener("click", () => closeSheet(lightbox));
    lightbox.addEventListener("click", (e) => { if (e.target === lightbox) closeSheet(lightbox); });

    const prevBtn = document.getElementById("lp-lightbox-prev");
    const nextBtn = document.getElementById("lp-lightbox-next");
    if (prevBtn) prevBtn.addEventListener("click", (e) => { e.stopPropagation(); showImageAt(currentIndex - 1); });
    if (nextBtn) nextBtn.addEventListener("click", (e) => { e.stopPropagation(); showImageAt(currentIndex + 1); });

    document.addEventListener("keydown", (e) => {
      if (!lightbox.classList.contains("lp-open")) return;
      if (e.key === "ArrowLeft") showImageAt(currentIndex - 1);
      if (e.key === "ArrowRight") showImageAt(currentIndex + 1);
      if (e.key === "Escape") closeSheet(lightbox);
    });

    let touchStartX = 0;
    lightbox.addEventListener("touchstart", (e) => { touchStartX = e.touches[0].clientX; }, { passive: true });
    lightbox.addEventListener("touchend", (e) => {
      const deltaX = e.changedTouches[0].clientX - touchStartX;
      if (Math.abs(deltaX) < 50) return;
      if (deltaX > 0) showImageAt(currentIndex - 1);
      else showImageAt(currentIndex + 1);
    }, { passive: true });
  }

    return `${months[d.getUTCMonth()]} ${d.getUTCDate()}, ${d.getUTCFullYear()}`;

    const div = document.createElement("div");
    div.className = "lp-mgrow-thumb lp-mgrow-thumb-empty";
    div.innerHTML = `<svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5"><rect x="3" y="3" width="18" height="18" rx="2"/><circle cx="8.5" cy="8.5" r="1.5"/><path d="M21 15l-5-5L5 21"/></svg>`;

function bindProducts(p, isOwner) {
  const section = document.getElementById("lp-products-section");
  const lockedSection = document.getElementById("lp-products-locked-section");
  const grid = document.getElementById("lp-products-grid");
  if (!section || !grid) return;

  fetch(`${WORKER_BASE}/api/profiles/${p.id}/products`, { credentials: "include" })
    .then((res) => res.json())
    .then(async (data) => {
      if (isOwner) showProductsManageTrigger(p, data.unlocked);

      if (!data.unlocked) {
        if (isOwner && lockedSection) renderOwnerLockedCard(lockedSection, p);
        return;
      }
      if (!data.products || !data.products.length) return; // stays hidden

      const products = data.products;
      let boostMap = {};
      try {
        const ids = products.map((prod) => prod.id).join(",");
        const boostRes = await fetch(`${WORKER_BASE}/api/profiles/${p.id}/boost-status?products=${encodeURIComponent(ids)}`);
        const boostData = await boostRes.json();
        boostMap = boostData.productBoosts || {};
      } catch (e) { /* boost badges are cosmetic — a failed fetch shouldn't block products rendering */ }

      renderProductLayout(grid, products, p, boostMap);
      section.hidden = false;
    })
    .catch((err) => console.error("Failed to load products:", err));
}

function renderOwnerLockedCard(lockedSection, p) {
  lockedSection.hidden = false;
  const titleEl = lockedSection.querySelector("h3");
  const descEl = lockedSection.querySelector("p");
  const shareBtn = document.getElementById("lp-products-locked-share-btn");

  fetch(`${WORKER_BASE}/api/profiles/${p.id}/referrals`, { credentials: "include" })
    .then((res) => (res.ok ? res.json() : { completed: [], pending: [] }))
    .then((refData) => {
      if (refData.pending && refData.pending.length && !(refData.completed && refData.completed.length)) {
        if (titleEl) titleEl.textContent = "Your referral has registered!";
        if (descEl) descEl.textContent = `${refData.pending[0].business_name} has joined using your link. Once they finish setting up their brand profile, your catalogue unlocks automatically.`;
      }
    })
    .catch(() => {});

  if (shareBtn && !shareBtn.dataset.wired) {
    shareBtn.dataset.wired = "true";
    shareBtn.addEventListener("click", () => {
      const refCode = p.referral_code || p.slug;
      const shareUrl = `${window.location.origin}${pagePath}?ref=${encodeURIComponent(refCode)}`;
      if (navigator.share) {
        navigator.share({ title: p.business_name, url: shareUrl }).catch(() => {});
      } else {
        navigator.clipboard.writeText(shareUrl).then(() => {
          const span = shareBtn.querySelector("span");
          const original = span.textContent;
          span.textContent = "Link Copied!";
          setTimeout(() => { span.textContent = original; }, 1800);
        });
      }
    });
  }
}

/* Builds the mixed layout: row 0 = single feed card; every row after
   uses PRODUCTS_ROW2_COLS-wide cards, still its own horizontal
   scroll-snap strip so a "row" can hold more than fits one screen. */
function renderProductLayout(grid, products, p, boostMap) {
  const inline = products.slice(0, PRODUCTS_INLINE_LIMIT);
  const overflow = products.slice(PRODUCTS_INLINE_LIMIT);

  let rowsHtml = "";

  if (inline.length) {
    // Row 1 — single full-width feed card (just the first product).
    rowsHtml += `
      <div class="lp-products-row-wrap">
        <div class="lp-products-row lp-products-row-single">
          ${feedCardHtml(inline[0], p, !!boostMap[inline[0].id])}
        </div>
      </div>
    `;

    // Remaining inline products — 2-column rows, each its own strip.
    const rest = inline.slice(1);
    for (let i = 0; i < rest.length; i += PRODUCTS_ROW2_COLS) {
      const rowProducts = rest.slice(i, i + PRODUCTS_ROW2_COLS);
      const isLastRow = (i + PRODUCTS_ROW2_COLS >= rest.length) && !overflow.length;
      rowsHtml += `
        <div class="lp-products-row-wrap">
          <div class="lp-products-row">
            ${rowProducts.map((prod) => compactCardHtml(prod, p, !!boostMap[prod.id])).join("")}
            ${(!isLastRow && i + PRODUCTS_ROW2_COLS >= rest.length && overflow.length) ? `
              <button type="button" class="lp-products-seeall-tile" data-see-all="true" aria-label="See all products">
                <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M9 18l6-6-6-6"/></svg>
                <span>See all (${products.length})</span>
              </button>` : ""}
          </div>
        </div>
      `;
    }
  }

  grid.innerHTML = rowsHtml;
  grid.dataset.allProducts = JSON.stringify(products);
  grid.dataset.boostMap = JSON.stringify(boostMap);
  wireProductCardActions(grid, p);
}

function wireProductCardActions(container, p) {
  container.querySelectorAll("[data-action='whatsapp']").forEach((btn) => {
    btn.addEventListener("click", (e) => {
      e.stopPropagation();
      const productName = btn.dataset.productName;
      const text = `Hi ${p.business_name}, I'm interested in "${productName}" — is it available?`;
      window.open(`https://wa.me/${cleanPhoneForLink(p.whatsapp_number)}?text=${encodeURIComponent(text)}`, "_blank");
    });
  });
  container.querySelectorAll("[data-action='call']").forEach((btn) => {
    btn.addEventListener("click", (e) => {
      e.stopPropagation();
      window.location.href = `tel:${cleanPhoneForLink(p.phone_number)}`;
    });
  });
  container.querySelectorAll("[data-see-all='true']").forEach((btn) => {
    btn.addEventListener("click", () => {
      const products = JSON.parse(container.dataset.allProducts || "[]");
      const boostMap = JSON.parse(container.dataset.boostMap || "{}");
      openProductsSeeAllSheet(products, p, boostMap);
    });
  });
  container.querySelectorAll("[data-action='zoom']").forEach((wrap) => {
    wrap.addEventListener("click", () => {
      const card = wrap.closest("[data-product-id]");
      const products = JSON.parse(container.dataset.allProducts || "[]");
      const index = products.findIndex((prod) => prod.id === card.dataset.productId);
      openProductLightbox(products, index >= 0 ? index : 0, p);
    });
  });
}

/* ---------------------------------------------------------------------
   "See all" bottom sheet — smart hybrid feed (2026-07-22 rebuild).
   Mostly a 2-column grid of compact cards (.lp-prodcard, same class
   used inline — one visual language, no second card style to keep in
   sync), but any product with a longer description automatically
   spans the full row as a feed card (.lp-prodfeed) so its text isn't
   crushed into a narrow column — a lightweight Facebook-feed accent
   inside an otherwise Play-Store-style grid. Every 6th compact card is
   also given a full-width feed treatment even without a long
   description, purely to break up the grid rhythm the way a real feed
   mixes card sizes. Tapping any card opens the SAME lightbox used
   elsewhere — a separate top-level overlay, so it always renders in
   front regardless of which sheet is open behind it.
   ------------------------------------------------------------------- */
const SEEALL_DESC_FEED_THRESHOLD = 70; // chars — beyond this, go full-width
const SEEALL_FEED_EVERY_N = 6;         // force a full-width break periodically

function openProductsSeeAllSheet(products, p, boostMap) {
  const host = getOverlayHost();
  let overlay = document.getElementById("lp-products-seeall-overlay");
  if (!overlay) {
    overlay = document.createElement("div");
    overlay.className = "lp-sheet-overlay";
    overlay.id = "lp-products-seeall-overlay";
    host.appendChild(overlay);
    overlay.addEventListener("click", (e) => { if (e.target === overlay) closeSheet(overlay); });
  }

  let compactStreak = 0;
  const itemsHtml = products.map((prod, i) => {
    const isBoosted = !!boostMap[prod.id];
    const longDesc = (prod.description || "").length > SEEALL_DESC_FEED_THRESHOLD;
    const dueForBreak = compactStreak >= SEEALL_FEED_EVERY_N;
    const useFeed = longDesc || dueForBreak;
    compactStreak = useFeed ? 0 : compactStreak + 1;
    return useFeed
      ? `<div class="lp-seeall-feed-item">${feedCardHtml(prod, p, isBoosted)}</div>`
      : `<div class="lp-seeall-grid-item">${compactCardHtml(prod, p, isBoosted)}</div>`;
  }).join("");

  overlay.innerHTML = `
    <div class="lp-sheet lp-sheet-tall">
      <div class="lp-sheet-top-bar">
        <div class="lp-sheet-handle"></div>
        <button class="lp-sheet-close" id="lp-products-seeall-close" type="button">
          <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>
        </button>
      </div>
      <h3 class="lp-sheet-title-only">All Products &amp; Services</h3>
      <div class="lp-seeall-feed" id="lp-products-seeall-feed">
        ${itemsHtml}
      </div>
    </div>
  `;
  document.getElementById("lp-products-seeall-close").addEventListener("click", () => closeSheet(overlay));

  const seeAllFeed = document.getElementById("lp-products-seeall-feed");
  seeAllFeed.dataset.allProducts = JSON.stringify(products);
  wireProductCardActions(seeAllFeed, p);

  openSheet(overlay);
}

function openProductLightbox(products, startIndex, p) {
  const lightbox = ensureProductLightboxEl();
  const img = document.getElementById("lp-product-lightbox-img");
  const nameEl = document.getElementById("lp-product-lightbox-name");
  const priceEl = document.getElementById("lp-product-lightbox-price");
  const descEl = document.getElementById("lp-product-lightbox-desc");
  const contactEl = document.getElementById("lp-product-lightbox-contact");
  const prevBtn = document.getElementById("lp-product-lightbox-prev");
  const nextBtn = document.getElementById("lp-product-lightbox-next");

  let currentIndex = startIndex;

  function updateNav() {
    prevBtn.style.visibility = currentIndex > 0 ? "visible" : "hidden";
    nextBtn.style.visibility = currentIndex < products.length - 1 ? "visible" : "hidden";
  }

  function showAt(index) {
    if (index < 0 || index >= products.length) return;
    currentIndex = index;
    const prod = products[index];
    if (prod.image_url) {
      img.src = prod.image_url;
      img.alt = prod.name;
      img.style.display = "";
      img.onerror = () => { img.style.display = "none"; };
    } else {
      img.style.display = "none";
    }
    nameEl.textContent = prod.name;
    priceEl.textContent = prod.price_display || "";
    priceEl.style.display = prod.price_display ? "" : "none";
    descEl.textContent = prod.description || "";
    descEl.style.display = prod.description ? "" : "none";
    contactEl.innerHTML = contactButtonsHtml(p, prod);
    wireProductCardActions(contactEl, p);
    updateNav();
  }

  showAt(currentIndex);
  openSheet(lightbox);

  const newPrev = prevBtn.cloneNode(true);
  prevBtn.parentNode.replaceChild(newPrev, prevBtn);
  const newNext = nextBtn.cloneNode(true);
  nextBtn.parentNode.replaceChild(newNext, nextBtn);
  newPrev.addEventListener("click", (e) => { e.stopPropagation(); showAt(currentIndex - 1); });
  newNext.addEventListener("click", (e) => { e.stopPropagation(); showAt(currentIndex + 1); });

  function keyHandler(e) {
    if (!lightbox.classList.contains("lp-open")) return;
    if (e.key === "ArrowLeft") showAt(currentIndex - 1);
    if (e.key === "ArrowRight") showAt(currentIndex + 1);
    if (e.key === "Escape") closeSheet(lightbox);
  }
  document.removeEventListener("keydown", lightbox._keyHandler || (() => {}));
  lightbox._keyHandler = keyHandler;
  document.addEventListener("keydown", keyHandler);

  let touchStartX = 0;
  lightbox.ontouchstart = (e) => { touchStartX = e.touches[0].clientX; };
  lightbox.ontouchend = (e) => {
    const diff = e.changedTouches[0].clientX - touchStartX;
    if (Math.abs(diff) > 50) diff > 0 ? showAt(currentIndex - 1) : showAt(currentIndex + 1);
  };
}

function showProductsManageTrigger(p, isUnlocked) {
  const unlockedBtn = document.getElementById("lp-pm-trigger-btn");
  const lockedBtn = document.getElementById("lp-pm-trigger-btn-locked");

  if (isUnlocked) {
    if (unlockedBtn && !unlockedBtn.dataset.wired) {
      unlockedBtn.dataset.wired = "true";
      unlockedBtn.hidden = false;
      unlockedBtn.addEventListener("click", () => openProductsManageModal(p));
    }
  } else {
    if (lockedBtn && !lockedBtn.dataset.wired) {
      lockedBtn.dataset.wired = "true";
      lockedBtn.hidden = false;
      lockedBtn.addEventListener("click", () => openProductsManageModal(p));
    }
  }

    searchInput.value = "";
    query = "";
    clearBtn.hidden = true;
    currentPage = 1;
    renderPage();
    searchInput.focus();
  });

  renderPage();
}

async function checkProductBoostStates(listEl, p, products) {
  try {
    const realProducts = products.filter((prod) => !prod.is_draft);
    if (!realProducts.length) return;
    const ids = realProducts.map((prod) => prod.id).join(",");
    const res = await fetch(`${WORKER_BASE}/api/profiles/${p.id}/boost-status?products=${encodeURIComponent(ids)}`);
    const data = await res.json();
    Object.entries(data.productBoosts || {}).forEach(([productId, boost]) => {
      if (!boost) return;
      const btn = listEl.querySelector(`[data-boost-product-id="${productId}"]`);
      if (btn) btn.classList.add("lp-mgrow-btn-boost-active");
    });
  } catch (e) { /* non-critical */ }
}

            <div class="lp-refitem-date">${formatReferralDate(ref.date)}</div>
          </div>
          <div class="lp-refitem-status ${ref.status === "done" ? "lp-refitem-status-done" : "lp-refitem-status-pending"}">
            ${ref.status === "done"
              ? `<svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3"><polyline points="20 6 9 17 4 12"/></svg> Completed`
              : "Pending"}
          </div>
        </div>
      `).join("");
    })
    .catch(() => { listEl.innerHTML = `<p class="lp-referrals-empty">Couldn't load your referrals right now.</p>`; });
}

function formatReferralDate(isoStr) {
  if (!isoStr) return "";
  try {
    const d = new Date(isoStr.includes("T") ? isoStr : isoStr.replace(" ", "T") + "Z");
    return d.toLocaleDateString(undefined, { year: "numeric", month: "short", day: "numeric" });
  } catch (e) {
    return "";
  }
}

/* ---------------------------------------------------------------------
   Add/Edit product sheet — same sheet handles both, so the fix for
   "no edit button" and the existing add flow share one code path.
   Image upload immediately creates (add mode) or updates (edit mode)
   a real product row the instant it finishes — this is what prevents
   an orphaned R2 file with no matching D1 row: the moment a photo is
   uploaded, a draft product already exists pointing at it.
   ------------------------------------------------------------------- */
function openProductFormSheet(p, existingProduct) {
  const isEdit = !!existingProduct;
  const host = getOverlayHost();
  let overlay = document.getElementById("lp-product-form-overlay");
  if (!overlay) {
    overlay = document.createElement("div");
    overlay.className = "lp-sheet-overlay";
    overlay.id = "lp-product-form-overlay";
    host.appendChild(overlay);
    overlay.addEventListener("click", (e) => { if (e.target === overlay) closeSheet(overlay); });
  }

  // In-memory state for this sheet session — tracks the product id
  // once a draft exists (either the existing product being edited, or
  // a fresh draft created the instant a photo uploads in add mode).
  let workingProductId = isEdit ? existingProduct.id : null;

  overlay.innerHTML = `
    <div class="lp-sheet">
      <div class="lp-sheet-top-bar">
        <div class="lp-sheet-handle"></div>
        <button class="lp-sheet-close" id="lp-product-form-close" type="button">
          <svg width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>
        </button>
      </div>
      <h3 class="lp-sheet-title-only">${isEdit ? "Edit Product or Service" : "Add a Product or Service"}</h3>
      <form id="lp-product-form">
        <div class="lp-product-form-card">
          <div class="lp-image-upload-box" id="lp-product-image-box">
            <div class="lp-image-upload-placeholder" id="lp-product-image-placeholder" style="${isEdit && existingProduct.image_url ? "display:none;" : ""}">
              <svg width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="17 8 12 3 7 8"/><line x1="12" y1="3" x2="12" y2="15"/></svg>
              <span>Tap to add a photo (optional)</span>
            </div>
            <img class="lp-image-upload-preview" id="lp-product-image-preview" style="${isEdit && existingProduct.image_url ? "" : "display:none;"}" src="${isEdit && existingProduct.image_url ? escAttr(existingProduct.image_url) : ""}" alt="">
            <div class="lp-image-upload-status" id="lp-product-image-status"></div>
          </div>
          <input type="file" accept="image/*" id="lp-product-file-input" style="display:none;">
          <input type="hidden" id="lp-product-image-url" value="${isEdit ? escAttr(existingProduct.image_url || "") : ""}">
        </div>
        <div class="lp-edit-field">
          <label class="lp-field-label">Product or service name</label>
          <input type="text" id="lp-product-name-input" class="lp-field-input" maxlength="80" required placeholder="e.g. Ankara Fabric Bundle" value="${isEdit && !existingProduct.is_draft ? escAttr(existingProduct.name) : ""}">
        </div>
        <div class="lp-edit-field">
          <label class="lp-field-label">Price (optional)</label>
          <input type="text" id="lp-product-price-input" class="lp-field-input" maxlength="40" placeholder="e.g. ₦15,000" value="${isEdit ? escAttr(existingProduct.price_display || "") : ""}">
        </div>
        <div class="lp-edit-field">
          <label class="lp-field-label">Description (optional)</label>
          <textarea id="lp-product-desc-input" class="lp-field-input lp-field-textarea" maxlength="500" placeholder="A short description of what you're offering">${isEdit ? escHtml(existingProduct.description || "") : ""}</textarea>
        </div>
        <div class="lp-edit-form-status" id="lp-product-form-status"></div>
        <button type="submit" class="lp-btn lp-btn-primary lp-submit-btn" id="lp-product-form-submit">
          <span>${isEdit ? "Save Changes" : "Add Product"}</span>
        </button>
      </form>
    </div>
  `;
  document.getElementById("lp-product-form-close").addEventListener("click", () => {
    closeSheet(overlay);
    refreshProductsManageModal(p); // refresh in case a draft was created but abandoned
  });
  openSheet(overlay);

  const fileBox = document.getElementById("lp-product-image-box");
  const fileInput = document.getElementById("lp-product-file-input");
  const preview = document.getElementById("lp-product-image-preview");
  const placeholder = document.getElementById("lp-product-image-placeholder");
  const statusEl = document.getElementById("lp-product-image-status");
  const hiddenUrl = document.getElementById("lp-product-image-url");

  fileBox.addEventListener("click", () => fileInput.click());
  fileInput.addEventListener("change", async () => {
    const file = fileInput.files[0];
    if (!file) return;
    statusEl.textContent = "Processing…";
    statusEl.className = "lp-image-upload-status lp-upload-active";
    try {
      const webpBlob = await compressImageToWebp(file, 800);
      statusEl.textContent = "Uploading…";
      const filename = buildSeoFilename(p.business_name, "product");
      const res = await fetch(`${WORKER_BASE}/api/upload-product-image?name=${encodeURIComponent(filename)}`, {
        method: "POST", credentials: "include",
        headers: { "content-type": "image/webp" }, body: webpBlob
      });
      const data = await res.json();
      if (!res.ok) throw new Error(data.error || "Upload failed");
      hiddenUrl.value = data.url;
      preview.src = data.url;
      preview.style.display = "block";
      placeholder.style.display = "none";
      statusEl.textContent = "";

      // Sync to D1 immediately — the moment the photo lands in R2, a
      // matching row exists so it's never orphaned, even if the owner
      // closes the sheet without tapping Save.
      if (!workingProductId) {
        const createRes = await fetch(`${WORKER_BASE}/api/products`, {
          method: "POST", credentials: "include",
          headers: { "content-type": "application/json" },
          body: JSON.stringify({ profile_id: p.id, image_url: data.url, is_draft: true })
        });
        const createData = await createRes.json();
        if (createRes.ok) workingProductId = createData.productId;
      } else {
        await fetch(`${WORKER_BASE}/api/products/${workingProductId}`, {
          method: "PATCH", credentials: "include",
          headers: { "content-type": "application/json" },
          body: JSON.stringify({ image_url: data.url })
        });
      }
    } catch (err) {
      statusEl.textContent = err.message;
      statusEl.className = "lp-image-upload-status lp-upload-error";
      showToast(err.message, "error");
    }
  });

  document.getElementById("lp-product-form").addEventListener("submit", async (e) => {
    e.preventDefault();
    const submitBtn = document.getElementById("lp-product-form-submit");
    const statusEl2 = document.getElementById("lp-product-form-status");
    submitBtn.disabled = true;
    submitBtn.innerHTML = `<span>${isEdit ? "Saving…" : "Saving…"}</span>`;

    const namedVal = document.getElementById("lp-product-name-input").value.trim();
    const priceVal = document.getElementById("lp-product-price-input").value.trim();
    const descVal = document.getElementById("lp-product-desc-input").value.trim();

    try {
      let res;
      if (workingProductId) {
        // Finalizes a draft (or edits an existing real product) —
        // same endpoint either way, since PATCH already clears
        // is_draft the moment a real name is supplied.
        res = await fetch(`${WORKER_BASE}/api/products/${workingProductId}`, {
          method: "PATCH", credentials: "include",
          headers: { "content-type": "application/json" },
          body: JSON.stringify({
            name: namedVal,
            price_display: priceVal || null,
            description: descVal || null
          })
        });
      } else {
        res = await fetch(`${WORKER_BASE}/api/products`, {
          method: "POST", credentials: "include",
          headers: { "content-type": "application/json" },
          body: JSON.stringify({
            profile_id: p.id,
            name: namedVal,
            price_display: priceVal || null,
            description: descVal || null,
            image_url: hiddenUrl.value || null
          })
        });
      }
      const data = await res.json();
      if (!res.ok) throw new Error(data.error || "Couldn't save this product.");

      closeSheet(overlay);
      showToast(isEdit || workingProductId ? "Product saved!" : "Product added!", "success");
      refreshProductsManageModal(p);
      bindProducts(p, true);
    } catch (err) {
      statusEl2.textContent = err.message;
      statusEl2.className = "lp-edit-form-status lp-edit-error";
      submitBtn.disabled = false;
      submitBtn.innerHTML = `<span>${isEdit ? "Save Changes" : "Add Product"}</span>`;
    }
  });
}

})();
